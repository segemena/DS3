299 def multinomial_combinations(n, k):
300  '''!
301  Return a list (2d numpy array) of all length k sequences of
302  non-negative integers n1, ..., nk such that n1 + ... + nk = n.
303  @param n: Number of jobs per snippet
304  @param k: Number of applications
305  @return List containing all combinations of n jobs and k applications that create a snippet
306  '''
307  bar_placements = itertools.combinations(range(1, n+k), k-1)
308  tmp = [(0,) + x + (n+k,) for x in bar_placements]
309  sequences = np.diff(tmp) - 1
310  return list(sequences[::-1])

312 def window_list(seq, n=2):
313  '''!
314  Return a sliding window_list (of width n) over data from the iterable, s -> (s0,s1,...s[n-1]), (s1,s2,...,sn), ...
315  '''
316  it = iter(seq)
317  result = tuple(islice(it, n))
318  if len(result) == n:
319  yield result
320  for elem in it:
321  result = result[1:] + (elem,)
322  yield result

494 def sed_inplace(filename, pattern, repl):
495  '''!
496  Perform the pure-Python equivalent of in-place `sed` substitution: e.g.,
497  `sed -i -e 's/'${pattern}'/'${repl}' "${filename}"`.
498  '''
499  pattern_compiled = re.compile(pattern)
500  with tempfile.NamedTemporaryFile(mode='w', delete=False) as tmp_file:
501  with open(filename) as src_file:
502  for line in src_file:
503  tmp_file.write(pattern_compiled.sub(repl, line))

507 def get_power_list_prediction(row, PEs, input_temperature):
508  '''!
509  Predict the power for all clusters based on the inputs
510  @param row: Dataset row (sample)
511  @param PEs: List of PEs
512  @param input_temperature: Baseline temperature
513  @return Power consumption prediction for all clusters
514  '''
515  power_list = []
516  for cluster in common.ClusterManager.cluster_list:
517  if cluster.type != 'MEM':
518  N_cores = 1
519  if cluster.type == 'LTL':
520  N_cores = int(row['N_little'])
521  elif cluster.type == 'BIG':
522  N_cores = int(row['N_big'])
523 
524  num_tasks = int(mean(cluster.snippet_num_tasks_list))
525  # num_tasks = int(max(cluster.snippet_num_tasks_list))
526  num_tasks = max(min(num_tasks, N_cores), 1)
527 
528  frequency = int(float(row['FREQ_PE_' + str(cluster.ID) + ' (GHz)']) * 1000)
529  voltage = DTPM_power_models.get_voltage_constant_mode(cluster.OPP, frequency)
530  static_power = DTPM_power_models.compute_static_power_dissipation(cluster.ID, input_temperature=input_temperature,
531  input_voltage=voltage)
532 
533  max_power_consumption, freq_threshold = DTPM_power_models.get_max_power_consumption(cluster, PEs, N_tasks=num_tasks,
534  N_cores=N_cores)
535  max_power_core = max_power_consumption / num_tasks
536  Cdyn_alpha = DTPM_power_models.compute_Cdyn_and_alpha(None, max_power_core, freq_threshold, OPP=cluster.OPP)
537  dynamic_power = DTPM_power_models.compute_dynamic_power_dissipation(frequency, voltage, Cdyn_alpha)
538 
539  total_power = static_power * N_cores + dynamic_power * min(num_tasks, N_cores)
540  power_list.append(total_power)
541  return power_list
